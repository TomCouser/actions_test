import UKFastAPI

[docs]class RecordManager(ICreateable, IGettable, IUpdateable, IDeleteable, IUrlCreator, IListable):
    """ Manager class for Record objects. """

    def __init__(self, parent):
        self._class = self.Record
        self._base = parent._manager._base

        if isinstance(parent, UKFastAPI.SafeDNS.template.TemplateManager.Template):
            self._url = UKFastAPI.SafeDNS.config.TEMPLATE_RECORD_URL.format(parent.id)
        if isinstance(parent, UKFastAPI.SafeDNS.zone.ZoneManager.Zone):
            self._url = UKFastAPI.SafeDNS.config.RECORD_URL.format(parent.name)

[docs]    def create(self, name: str, type: str, content: str, **kwargs):
        """
        Used to create a Record within a UKFast Zone instance.

        Parameters:
        name (str): The name for the Zone.
        description (string, optional): The description of the Zone.

        Returns:
        Zone: The newly created UKFast Zone.
        """

        kwargs['body'] = dict({
            'name': name,
            'type': type,
            'content': content,
        }, **kwargs)

        record_id = super()._create(self._url, **kwargs).id
        return super()._get(super()._get_url(record_id))

        """
        Used to get a pre-existing Record from a UKFast Zone instance.

        Parameters:
        id (str): The id of the Record.

        Returns:
        Record: The matching Record.
        """
        return super()._get(super()._get_url(id), **kwargs)


    def _save(self, record):
        return super()._update('{}/{}'.format(self._url, record.id),
                               **{'body': {
                                   'name': record.name,
                                   'type': record.type,
                                   'content': record.content,
                                   'priority': record.priority,
                                   'ttl': record.ttl
                               }})
