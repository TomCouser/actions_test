    """ Manager class for Template objects. """
    max_name_len = 255

    def __init__(self, safedns):
        self._class = self.Template
        self._base = safedns._base
        self._url = config.TEMPLATE_URL


[docs]    def create(self, name: str, **kwargs):
        """
        Used to create a UKFast Template instance.

        Parameters:
        name (str): The name for the Teplate.
        default (str, optional): The description of the Template.

        Returns:
        Template: The newly created UKFast Template.
        """

        if len(name) > self.max_name_len:
            raise exceptions.UKFastSDKException(
                'Template names cannot be more than 255 characters.')

        kwargs['body'] = {'name': name, 'default': kwargs.get('default', False)}
        template = super()._create(self._url, **kwargs)
        return super()._get(super()._get_url(template.id))
