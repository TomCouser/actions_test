[docs]    def create(self, name: str, **kwargs):
        """
        Used to create a UKFast Zone instance.

        Parameters:
        name (str): The name for the Zone.
        description (string, optional): The description of the Zone.

        Returns:
        Zone: The newly created UKFast Zone.
        """
        if not isinstance(name, str):
            raise exceptions.UKFastSDKException('name must be an str.')

        kwargs['body'] = {'name': name, 'description': kwargs.get('description', '')}
        zone = super()._create(self._url, **kwargs)
        return super()._get(super()._get_url(zone.name))
