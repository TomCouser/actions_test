[docs]    def create(self, user_id: int, contents: str, **kwargs):
        """
        Used to create a Note within a UKFast Zone instance.

        Parameters:
        user_id (str): The user id for the note.
        note (str): The contents of the note.

        Returns:
        Note: The newly created UKFast Note.
        """

        if user_id > 2147483647:
            raise exceptions.UKFastSDKException('user_id must not exceed max int.')
        if len(contents) > 65536:
            raise exceptions.UKFastSDKException('content has exceeded max length of 65536.')

        kwargs['body'] = {
            'contact_id': user_id,
            'notes': contents,
        }
        note = super()._create(self._url, **kwargs)
        return super()._get(super()._get_url(note.id))
